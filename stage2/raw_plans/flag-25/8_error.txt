residuals AS (
    SELECT 
        f.category,
        f.department,
        f.amount,
        (f.processed_date - f.opened_at) as actual_days,
        ROUND((p.intercept + p.slope * f.amount)::numeric, 2) as predicted_days,
        ROUND(((f.processed_date - f.opened_at) - (p.intercept + p.slope * f.amount))::numeric, 2) as residual
    FROM flag_25 f, params p
    WHERE f.processed_date IS NOT NULL AND f.opened_at IS NOT NULL
)
SELECT 
    category,
    COUNT(*) as n,
    ROUND((AVG(residual))::numeric, 3) as avg_residual,
    ROUND((STDDEV(residual))::numeric, 3) as stddev_residual,
    ROUND((MIN(residual))::numeric, 3) as min_residual,
    ROUND((MAX(residual))::numeric, 3) as max_residual
FROM residuals
GROUP BY category
ORDER BY avg_residual DESC

residuals AS (
    SELECT 
        f.department,
        f.category,
        f.amount,
        (f.processed_date - f.opened_at) as actual_days,
        (p.intercept + p.slope * f.amount) as predicted_raw,
        (f.processed_date - f.opened_at) - (p.intercept + p.slope * f.amount) as residual
    FROM flag_25 f, params p
    WHERE f.processed_date IS NOT NULL AND f.opened_at IS NOT NULL
)
SELECT 
    department,
    category,
    COUNT(*) as n,
    ROUND((AVG(residual))::numeric, 3) as avg_residual,
    ROUND((STDDEV(residual))::numeric, 3) as stddev_residual
FROM residuals
GROUP BY department, category
HAVING COUNT(*) >= 5
ORDER BY avg_residual DESC
